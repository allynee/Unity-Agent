private Object3D lamp;

public void CreateLamp()
{
    // Default spawn lamp at the user's feet
    Vector3D userFeetPosition = GetUsersFeetPosition();
    
    // Create the lamp
    lamp = CreateObject("NewLamp", "Lamp", userFeetPosition, new Vector3D(0, 0, 0));
}
