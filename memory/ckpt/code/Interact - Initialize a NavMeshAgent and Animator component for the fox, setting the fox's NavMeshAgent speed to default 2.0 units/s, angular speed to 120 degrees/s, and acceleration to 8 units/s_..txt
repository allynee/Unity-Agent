Instruction:
Initialize a NavMeshAgent and Animator component for the fox, setting the fox's NavMeshAgent speed to default 2.0 units/s, angular speed to 120 degrees/s, and acceleration to 8 units/s_.

Category:
Interact

Code:
private void InitializeComponents()
  {
  if (foxGameObject == null)
  {
  Debug.LogError("Fox GameObject is not assigned!");
  return;
  }
 

  navMeshAgent = foxGameObject.GetComponent<NavMeshAgent>();
  if (navMeshAgent == null)
  {
  navMeshAgent = foxGameObject.AddComponent<NavMeshAgent>();
  }
  
  foxAnimator = foxGameObject.GetComponent<Animator>();
  
  // Set default values
  navMeshAgent.speed = 2.0f;
  navMeshAgent.angularSpeed = 120f;
  navMeshAgent.acceleration = 8f;
  }
